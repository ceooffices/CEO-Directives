// ======================================== // RENDER PROCESS — HORIZONTAL TIMELINE // ======================================== var currentOpenStage = 1; // Track which stage panel is open function renderProcess(data) { if (!data) return; processData = data; document.getElementById("processTarget").textContent = data.targetDate; var doneCount = 0; for (var i = 0; i < data.steps.length; i++) { var stageEl = document.getElementById("stage" + (i + 1)); var dotEl = stageEl ? stageEl.querySelector(".v-stage-dot") : null; if (!stageEl) continue; var s = data.steps[i]; stageEl.classList.remove("stage-done", "stage-active"); if (s.status === "done") { stageEl.classList.add("stage-done"); if (dotEl) dotEl.textContent = "✓"; doneCount++; } else if (s.status === "active") { stageEl.classList.add("stage-active"); if (dotEl) dotEl.textContent = String(i + 1); } else { if (dotEl) dotEl.textContent = String(i + 1); } } // Update timeline bar fill var fillPct = data.steps.length > 0 ? (doneCount / data.steps.length) * 100 : 0; var barFill = document.getElementById("timelineBarFill"); if (barFill) barFill.style.width = fillPct + "%"; // Auto-open active stage panel for (var i = 0; i < data.steps.length; i++) { if (data.steps[i].status === "active") { toggleStage(i + 1); break; } } } // ======================================== // TOGGLE STAGE (horizontal timeline) // ======================================== function toggleStage(num) { var panel = document.getElementById("tlExpandPanel"); if (!panel) return; // If same stage clicked, toggle close if (currentOpenStage === num && panel.classList.contains("open")) { panel.classList.remove("open", "panel-active", "panel-done"); // Hide all panel contents for (var j = 1; j <= 4; j++) { var pc = document.getElementById("panelContent" + j); if (pc) pc.style.display = "none"; var st = document.getElementById("stage" + j); if (st) st.classList.remove("open"); } currentOpenStage = 0; return; } // Hide all, show selected for (var j = 1; j <= 4; j++) { var pc = document.getElementById("panelContent" + j); if (pc) pc.style.display = j === num ? "block" : "none"; var st = document.getElementById("stage" + j); if (st) { if (j === num) st.classList.add("open"); else st.classList.remove("open"); } } // Set panel state panel.classList.remove("panel-active", "panel-done"); var stageEl = document.getElementById("stage" + num); if (stageEl && stageEl.classList.contains("stage-active")) panel.classList.add("panel-active"); else if (stageEl && stageEl.classList.contains("stage-done")) panel.classList.add("panel-done"); panel.classList.add("open"); currentOpenStage = num; // Update expand head num color var headNum = panel.querySelector( "#panelContent" + num + " .tl-expand-head-num", ); if (headNum) { if (stageEl && stageEl.classList.contains("stage-done")) headNum.style.background = "var(--c-approved)"; else headNum.style.background = "var(--c-total)"; } // Smooth scroll to panel on mobile so user sees expanded content setTimeout(function() { var stageHead = document.getElementById("stage" + num); if (stageHead) { stageHead.scrollIntoView({ behavior: 'smooth', block: 'start' }); } }, 150); } // ======================================== // AUTO-CHECK PROCESS CHECKLISTS (LOGIC CHÍNH XÁC) // ======================================== function updateChecklists(data) { if (!data || !data.items) { // Không có dữ liệu → tất cả checklist chưa hoàn thành resetAllChecklists(); updateStepProgress(); return; } var items = data.items; var total = items.length; var approved = 0, pending = 0, rejected = 0, hasOrder = 0; for (var i = 0; i < items.length; i++) { if (items[i].status === "Duyệt") approved++; if (items[i].status === "Chờ duyệt") pending++; if (items[i].status === "Từ chối") rejected++; if (items[i].thuTu && parseInt(items[i].thuTu) > 0) hasOrder++; } // --- STAGE 1: Tiếp nhận đăng ký --- // formOpen: chỉ true khi form thực sự đang mở (kiểm tra qua có đăng ký nào không) setCheck("formOpen", total > 0); // Có ít nhất 1 đăng ký → form đang hoạt động setCheck("hasRegistrations", total > 0); // Kiểm tra tất cả bộ phận bắt buộc đã đăng ký var deptRows = document.querySelectorAll("#deptTableBody tr"); var allDeptDone = true; var deptCount = 0; var anyDeptExists = false; for (var d = 0; d < deptRows.length; d++) { var statusCell = deptRows[d].querySelector(".dept-badge"); if (!statusCell) continue; anyDeptExists = true; deptCount++; // "Chưa đăng ký" = chưa hoàn thành if (statusCell.classList.contains("dept-missing")) { allDeptDone = false; } } // allDeptDone chỉ true khi: có bộ phận trong danh sách VÀ tất cả đều đã đăng ký setCheck("allDeptDone", anyDeptExists && deptCount > 0 && allDeptDone); // --- STAGE 2: Phê duyệt --- // hasPending: true khi có đăng ký VÀ không còn chờ duyệt nào setCheck("hasPending", total > 0 && pending === 0); // hasOrder: true khi có nội dung được duyệt VÀ tất cả đã được xếp thứ tự setCheck("hasOrder", approved > 0 && hasOrder >= approved); // resultSent: cần backend xác nhận (kiểm tra cột emailSent) var allResultSent = approved > 0; for (var i = 0; i < items.length; i++) { if ((items[i].status === "Duyệt" || items[i].status === "Từ chối" || items[i].status === "Hoãn") && !items[i].emailSent) { allResultSent = false; break; } } setCheck("resultSent", allResultSent); // --- STAGE 3: Xếp lịch --- setCheck("scheduleReady", approved > 0 && hasOrder >= approved); setCheck("scheduleSent", false); // Manual: cần backend confirm // --- STAGE 4: Biên bản --- setCheck("meetingDone", false); // Manual setCheck("minutesSent", false); // Manual // Update progress count trên step bar updateStepProgress(); } function resetAllChecklists() { var allItems = document.querySelectorAll('.check-item'); for (var i = 0; i < allItems.length; i++) { allItems[i].classList.remove('checked'); var icon = allItems[i].querySelector('.check-icon'); if (icon) icon.textContent = '○'; } } function setCheck(key, isDone) { var el = document.querySelector( '.check-item[data-check="' + key + '"]', ); if (!el) return; var icon = el.querySelector(".check-icon"); if (isDone) { el.classList.add("checked"); if (icon) icon.textContent = "✓"; } else { el.classList.remove("checked"); if (icon) icon.textContent = "○"; } } function updateStepProgress() { for (var s = 1; s <= 4; s++) { var checklist = document.getElementById("checklist" + s); if (!checklist) continue; var allItems = checklist.querySelectorAll(".check-item"); var doneItems = checklist.querySelectorAll(".check-item.checked"); var progEl = document.getElementById("stageProgress" + s); if (progEl && allItems.length > 0) { progEl.textContent = doneItems.length + "/" + allItems.length; } // Auto mark stage done/active var stageEl = document.getElementById("stage" + s); if ( stageEl && doneItems.length === allItems.length && allItems.length > 0 ) { stageEl.classList.add("stage-done"); stageEl.classList.remove("stage-active"); } } }
