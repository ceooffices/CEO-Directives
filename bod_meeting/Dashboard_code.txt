    <script>
      // ========================================
      // STATE
      // ========================================
      var currentData = null;
      var activeFilter = "all";

      // ========================================
      // INITIALIZATION
      // ========================================
      function init() {
        google.script.run
          .withSuccessHandler(function (dates) {
            var select = document.getElementById("dateSelect");
            select.innerHTML = '<option value="">📅 Tất cả ngày họp</option>';

            dates.forEach(function (d) {
              var opt = document.createElement("option");
              opt.value = d.search;
              opt.textContent = d.display;
              select.appendChild(opt);
            });

            // Auto-chọn ngày gần nhất
            if (dates.length > 0) {
              select.value = dates[0].search;
            }
            loadData();
          })
          .withFailureHandler(function (err) {
            document.getElementById("tableBody").innerHTML =
              '<tr><td colspan="7"><div class="empty-state">' +
              '<div class="empty-icon">⚠️</div>' +
              "<p>Không thể kết nối. Hãy kiểm tra kết nối mạng.</p>" +
              "</div></td></tr>";
          })
          .getAvailableDates();
      }

      // ========================================
      // LOAD DATA
      // ========================================
      function loadData() {
        var searchDate = document.getElementById("dateSelect").value;

        // Show loading
        document.getElementById("tableBody").innerHTML =
          '<tr><td colspan="7"><div class="loading-state">' +
          '<div class="spinner"></div>' +
          "<p>Đang tải dữ liệu...</p></div></td></tr>";

        google.script.run
          .withSuccessHandler(function (data) {
            currentData = data;
            activeFilter = "all";
            renderCards(data);
            renderTable(data.items);
          })
          .withFailureHandler(function (err) {
            document.getElementById("tableBody").innerHTML =
              '<tr><td colspan="7"><div class="empty-state">' +
              '<div class="empty-icon">❌</div>' +
              "<p>Lỗi tải dữ liệu: " +
              (err.message || "Không xác định") +
              "</p>" +
              "</div></td></tr>";
          })
          .getDashboardData(searchDate);
      }

      // ========================================
      // RENDER STAT CARDS
      // ========================================
      function renderCards(data) {
        var grid = document.getElementById("cardsGrid");

        var cards = [
          {
            key: "all",
            label: "Tổng đăng ký",
            value: data.total,
            css: "card-total",
            hint: "Nhấn để xem tất cả",
          },
          {
            key: "Duyệt",
            label: "Đã duyệt",
            value: data.approved,
            css: "card-approved",
            hint: "Nhấn để lọc",
          },
          {
            key: "Chờ duyệt",
            label: "Chờ duyệt",
            value: data.pending,
            css: "card-pending",
            hint: "Nhấn để lọc",
          },
          {
            key: "Từ chối",
            label: "Từ chối",
            value: data.rejected,
            css: "card-rejected",
            hint: "Nhấn để lọc",
          },
          {
            key: "Hoãn",
            label: "Tạm hoãn",
            value: data.postponed,
            css: "card-postponed",
            hint: "Nhấn để lọc",
          },
        ];

        grid.innerHTML = cards
          .map(function (card) {
            var isActive = activeFilter === card.key ? " active" : "";
            return (
              '<div class="stat-card ' +
              card.css +
              isActive +
              '" ' +
              "onclick=\"filterBy('" +
              card.key +
              "')\" " +
              'role="button" tabindex="0" ' +
              'aria-label="Lọc theo ' +
              card.label +
              '">' +
              '<div class="card-number">' +
              card.value +
              "</div>" +
              '<div class="card-label">' +
              card.label +
              "</div>" +
              '<div class="card-hint">👆 ' +
              card.hint +
              "</div>" +
              "</div>"
            );
          })
          .join("");
      }

      // ========================================
      // FILTER BY STATUS
      // ========================================
      function filterBy(key) {
        activeFilter = key;
        renderCards(currentData);

        var filteredItems;
        var badgeEl = document.getElementById("filterBadge");

        if (key === "all") {
          filteredItems = currentData.items;
          badgeEl.innerHTML = "";
        } else {
          filteredItems = currentData.items.filter(function (item) {
            return item.status === key;
          });

          var colorMap = {
            Duyệt: { bg: "#D1FAE5", text: "#065F46" },
            "Chờ duyệt": { bg: "#FEF3C7", text: "#92400E" },
            "Từ chối": { bg: "#FEE2E2", text: "#991B1B" },
            Hoãn: { bg: "#EDE9FE", text: "#5B21B6" },
          };
          var colors = colorMap[key] || { bg: "#E2E8F0", text: "#475569" };

          badgeEl.innerHTML =
            '<span class="filter-badge" style="background:' +
            colors.bg +
            ";color:" +
            colors.text +
            '">' +
            "🔍 " +
            key +
            " — " +
            filteredItems.length +
            " kết quả" +
            ' <span class="clear-btn" onclick="event.stopPropagation(); filterBy(\'all\')" title="Bỏ lọc">✕</span>' +
            "</span>";
        }

        renderTable(filteredItems);
      }

      // ========================================
      // RENDER DATA TABLE
      // ========================================
      function renderTable(items) {
        var tbody = document.getElementById("tableBody");

        if (!items || items.length === 0) {
          tbody.innerHTML =
            '<tr><td colspan="7"><div class="empty-state">' +
            '<div class="empty-icon">📭</div>' +
            "<p>Không có đăng ký nào" +
            (activeFilter !== "all"
              ? ' cho trạng thái "' + activeFilter + '"'
              : "") +
            "</p></div></td></tr>";
          return;
        }

        var statusClassMap = {
          Duyệt: "status-approved",
          "Từ chối": "status-rejected",
          Hoãn: "status-postponed",
          "Chờ duyệt": "status-pending",
        };

        var statusIconMap = {
          Duyệt: "✅",
          "Từ chối": "❌",
          Hoãn: "⏸️",
          "Chờ duyệt": "⏳",
        };

        tbody.innerHTML = items
          .map(function (item, index) {
            var statusClass = statusClassMap[item.status] || "status-pending";
            var statusIcon = statusIconMap[item.status] || "⏳";
            var emailClass = item.daGuiEmail ? "email-sent" : "email-not-sent";
            var emailIcon = item.daGuiEmail ? "✔︎" : "○";
            var emailText = item.daGuiEmail ? "Đã gửi" : "Chưa gửi";

            return (
              '<tr class="row-highlight">' +
              '<td style="font-weight:800;text-align:center;color:#94A3B8;font-size:15px">' +
              (index + 1) +
              "</td>" +
              '<td style="font-weight:700">' +
              escapeHtml(item.hoTen) +
              "</td>" +
              '<td><span style="font-weight:700;color:var(--c-total)">' +
              escapeHtml(item.boPhan) +
              "</span></td>" +
              '<td><div class="cell-content">' +
              escapeHtml(item.noiDung) +
              "</div></td>" +
              '<td style="text-align:center;font-weight:700">' +
              escapeHtml(item.thoiLuong) +
              "</td>" +
              '<td><span class="status-badge ' +
              statusClass +
              '">' +
              statusIcon +
              " " +
              item.status +
              "</span></td>" +
              '<td><span class="email-badge ' +
              emailClass +
              '">' +
              emailIcon +
              " " +
              emailText +
              "</span></td>" +
              "</tr>"
            );
          })
          .join("");
      }

      // ========================================
      // UTILITY
      // ========================================
      function escapeHtml(str) {
        if (!str) return "";
        var div = document.createElement("div");
        div.textContent = str;
        return div.innerHTML;
      }

      // Keyboard support for cards
      document.addEventListener("keydown", function (e) {
        if (
          e.target.classList &&
          e.target.classList.contains("stat-card") &&
          (e.key === "Enter" || e.key === " ")
        ) {
          e.preventDefault();
          e.target.click();
        }
      });

      // ========================================
      // START
      // ========================================
      init();
    </script>
